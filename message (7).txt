--!nocheck
--[[
    Standalone "Hit Boxes + Reach" — with toggle for hitboxes (Backquote key) + persistent status indicator.
    
    - NO keybind for reach: it's always on.
    - Hitboxes start ON. Press Backquote (`) to toggle them OFF/ON (Player mode only).
    - A small light appears in the bottom‑right corner:
         green = hitboxes ON (Player mode, togglable)
         red   = hitboxes OFF
         yellow = Sword mode (hitboxes always on, toggle disabled)
    - Placement/break reach intentionally removed; only sword reach + bypass.
]]

local Players = game:GetService('Players')
local ReplicatedStorage = game:GetService('ReplicatedStorage')
local UserInputService = game:GetService('UserInputService')

local lplr = Players.LocalPlayer

--============================ CONFIG ============================--
local CONFIG = {
    -- ===== Hit Boxes =====
    HitboxMode = 'Player',   -- 'Player' (welded boxes) or 'Sword' (swing-region expand)
    HitboxExpand = 14.4,     -- 0 .. 14.4 studs
    HitboxTransparency = 1,  -- 1 = invisible (original). Lower it to SEE the boxes.

    -- ===== Reach =====
    SwordReach = true,       -- applied distance = SwordRange + 2 (off = 14.4). Keep <= 18.
    SwordRange = 14.89,
}
--===============================================================--

local hitboxObjects = {}          -- [player] = Part
local hitboxPlayerConns = {}      -- [player] = CharacterAdded connection
local hitboxesActive = true       -- toggle state
local hitboxAddedConn = nil       -- PlayerAdded connection
local hitboxRemovingConn = nil    -- PlayerRemoving connection

local indicatorFrame = nil        -- the small status light
local playerGuiConn = nil         -- connection for PlayerGui being added

----------------------------------------------------------------
-- DEPENDENCY RESOLUTION (no Vape)
----------------------------------------------------------------
local function getSwordController()
    if getgenv and getgenv().bedwars then
        local ok, sc = pcall(function()
            return getgenv().bedwars.SwordController
        end)
        if ok and sc then
            return sc
        end
    end
    if typeof(getgc) == 'function' then
        for _, obj in getgc(true) do
            if type(obj) == 'table' then
                local ok, fn = pcall(rawget, obj, 'swingSwordInRegion')
                if ok and type(fn) == 'function' then
                    return obj
                end
            end
        end
    end
    return nil
end

local function getCombatConstant()
    if getgenv and getgenv().bedwars then
        local ok, cc = pcall(function()
            return getgenv().bedwars.CombatConstant
        end)
        if ok and cc then
            return cc
        end
    end
    local ok, cc = pcall(function()
        return require(ReplicatedStorage.TS.combat['combat-constant']).CombatConstant
    end)
    if ok and cc then
        return cc
    end
    return nil
end

----------------------------------------------------------------
-- HIT BOXES (Player mode – togglable)
----------------------------------------------------------------
local function applyHitboxRegion()
    if typeof(debug) ~= 'table' or typeof(debug.setconstant) ~= 'function' then
        warn('[HitBoxes] Sword mode needs debug.setconstant — not supported. Set HitboxMode = "Player".')
        return
    end
    local sc = getSwordController()
    if not sc then
        warn('[HitBoxes] SwordController not found. Set HitboxMode = "Player".')
        return
    end
    pcall(debug.setconstant, sc.swingSwordInRegion, 6, (CONFIG.HitboxExpand / 3))
end

local function createHitbox(plr, char)
    if plr == lplr or not char then
        return
    end
    local root = char:FindFirstChild('HumanoidRootPart')
    if not root then
        return
    end

    -- replace any existing box
    if hitboxObjects[plr] then
        hitboxObjects[plr]:Destroy()
        hitboxObjects[plr] = nil
    end

    local hitbox = Instance.new('Part')
    hitbox.Size = Vector3.new(3, 6, 3) + Vector3.one * (CONFIG.HitboxExpand / 5)
    hitbox.CFrame = root.CFrame
    hitbox.CanCollide = false
    hitbox.Massless = true
    hitbox.Transparency = CONFIG.HitboxTransparency
    hitbox.Color = Color3.fromRGB(255, 40, 90)
    hitbox.Material = Enum.Material.SmoothPlastic
    local weld = Instance.new('Motor6D')
    weld.Part0 = hitbox
    weld.Part1 = root
    weld.Parent = hitbox
    hitbox.Parent = char
    hitboxObjects[plr] = hitbox
end

local function onCharacterAdded(plr, char)
    local root = char:WaitForChild('HumanoidRootPart', 10)
    local hum = char:WaitForChild('Humanoid', 10)
    if not (root and hum) or char.Parent == nil then
        return
    end
    createHitbox(plr, char)
    hum.Died:Once(function()
        if hitboxObjects[plr] then
            hitboxObjects[plr]:Destroy()
            hitboxObjects[plr] = nil
        end
    end)
end

local function trackPlayer(plr)
    if plr == lplr then return end
    -- disconnect any old connection for this player first
    if hitboxPlayerConns[plr] then
        hitboxPlayerConns[plr]:Disconnect()
    end
    hitboxPlayerConns[plr] = plr.CharacterAdded:Connect(function(char)
        onCharacterAdded(plr, char)
    end)
    if plr.Character then
        task.spawn(onCharacterAdded, plr, plr.Character)
    end
end

local function disableHitboxes()
    if not hitboxesActive then return end
    hitboxesActive = false

    -- disconnect all player tracking
    for plr, conn in pairs(hitboxPlayerConns) do
        conn:Disconnect()
        hitboxPlayerConns[plr] = nil
    end
    if hitboxAddedConn then
        hitboxAddedConn:Disconnect()
        hitboxAddedConn = nil
    end
    if hitboxRemovingConn then
        hitboxRemovingConn:Disconnect()
        hitboxRemovingConn = nil
    end

    -- destroy all hitboxes
    for plr, part in pairs(hitboxObjects) do
        part:Destroy()
        hitboxObjects[plr] = nil
    end

    -- update status light
    if indicatorFrame then
        indicatorFrame.BackgroundColor3 = Color3.fromRGB(255, 0, 0)   -- red = off
    end

    print('[HitBoxes] Hitboxes turned OFF')
end

local function enableHitboxes()
    if hitboxesActive then return end
    hitboxesActive = true

    -- start tracking existing players
    for _, plr in ipairs(Players:GetPlayers()) do
        trackPlayer(plr)
    end

    -- set up new player listeners
    if not hitboxAddedConn then
        hitboxAddedConn = Players.PlayerAdded:Connect(function(plr)
            if hitboxesActive then
                trackPlayer(plr)
            end
        end)
    end
    if not hitboxRemovingConn then
        hitboxRemovingConn = Players.PlayerRemoving:Connect(function(plr)
            if hitboxObjects[plr] then
                hitboxObjects[plr]:Destroy()
                hitboxObjects[plr] = nil
            end
            if hitboxPlayerConns[plr] then
                hitboxPlayerConns[plr]:Disconnect()
                hitboxPlayerConns[plr] = nil
            end
        end)
    end

    -- update status light
    if indicatorFrame then
        indicatorFrame.BackgroundColor3 = Color3.fromRGB(0, 255, 0)   -- green = on
    end

    print('[HitBoxes] Hitboxes turned ON')
end

local function toggleHitboxes()
    if CONFIG.HitboxMode ~= 'Player' then
        warn('[HitBoxes] Toggle only works in Player mode. Set HitboxMode = "Player".')
        return
    end
    if hitboxesActive then
        disableHitboxes()
    else
        enableHitboxes()
    end
end

local function startHitboxes()
    if CONFIG.HitboxMode == 'Sword' then
        applyHitboxRegion()
        print('[HitBoxes] Sword mode active (toggle disabled)')
        return
    end
    -- Player mode: start enabled
    enableHitboxes()
end

----------------------------------------------------------------
-- REACH (sword only; always on)
----------------------------------------------------------------
local function applySwordReach()
    local cc = getCombatConstant()
    if not cc then
        warn('[Reach] CombatConstant not found — are you in a BedWars round?')
        return
    end
    local target = (CONFIG.SwordReach and (CONFIG.SwordRange + 2)) or 14.4
    local ok, err = pcall(function()
        cc.RAYCAST_SWORD_CHARACTER_DISTANCE = target
    end)
    if not ok then
        warn('[Reach] Failed to write reach constant (table may be frozen): ' .. tostring(err))
        return
    end
    print('[Reach] Client raycast distance = ' .. tostring(cc.RAYCAST_SWORD_CHARACTER_DISTANCE) .. ' studs.')
end

-- Server‑side validation bypass (required for sword reach hits to register)
local attackHookInstalled = false
local function installAttackHook()
    if attackHookInstalled or (getgenv and getgenv().__HitboxReachAttackHook) then
        return
    end
    if typeof(hookmetamethod) ~= 'function' or typeof(getnamecallmethod) ~= 'function' then
        warn('[Reach] Executor lacks hookmetamethod/getnamecallmethod – server will reject hits past 14.4 studs.')
        return
    end
    local wrap = (typeof(newcclosure) == 'function') and newcclosure or function(f) return f end

    local oldNamecall
    oldNamecall = hookmetamethod(game, '__namecall', wrap(function(self, ...)
        local method = getnamecallmethod()
        if method == 'FireServer' or method == 'InvokeServer' then
            local packet = (...)
            if type(packet) == 'table' then
                local validate = rawget(packet, 'validate')
                if type(validate) == 'table' then
                    local sp = rawget(validate, 'selfPosition')
                    local tp = rawget(validate, 'targetPosition')
                    if
                        type(sp) == 'table'
                        and type(tp) == 'table'
                        and typeof(rawget(sp, 'value')) == 'Vector3'
                        and typeof(rawget(tp, 'value')) == 'Vector3'
                    then
                        local selfpos = rawget(sp, 'value')
                        local targetpos = rawget(tp, 'value')
                        local dist = (selfpos - targetpos).Magnitude
                        if dist > 14.399 then
                            validate.raycast = rawget(validate, 'raycast') or {}
                            sp.value = selfpos + CFrame.lookAt(selfpos, targetpos).LookVector * (dist - 14.399)
                        end
                    end
                end
            end
        end
        return oldNamecall(self, ...)
    end))

    attackHookInstalled = true
    if getgenv then
        getgenv().__HitboxReachAttackHook = true
    end
    print('[Reach] Server-side hit validation bypass installed.')
end

local function startReach()
    applySwordReach()
    installAttackHook()
end

-- Re-apply sword values after editing CONFIG live
local function refresh()
    applySwordReach()
    if CONFIG.HitboxMode == 'Sword' then
        applyHitboxRegion()
    end
end

----------------------------------------------------------------
-- STATUS INDICATOR (bottom‑right light) – PERSISTENT VERSION
----------------------------------------------------------------
local function createIndicator()
    local playerGui = lplr:FindFirstChild("PlayerGui")
    if not playerGui then
        -- Wait for PlayerGui to appear (e.g., after respawn if game recreates it)
        playerGui = lplr:WaitForChild("PlayerGui", 10)
        if not playerGui then return end
    end

    -- Remove any existing indicator GUI to avoid duplicates
    local oldGui = playerGui:FindFirstChild("HitboxIndicator")
    if oldGui then oldGui:Destroy() end

    local gui = Instance.new("ScreenGui")
    gui.Name = "HitboxIndicator"
    gui.Parent = playerGui

    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0, 10, 0, 10)
    frame.Position = UDim2.new(1, -14, 1, -14)
    frame.AnchorPoint = Vector2.new(1, 1)
    
    -- Set color based on current mode/state
    if CONFIG.HitboxMode == 'Sword' then
        frame.BackgroundColor3 = Color3.fromRGB(255, 255, 0)   -- yellow: Sword mode (toggle disabled)
    else
        frame.BackgroundColor3 = hitboxesActive and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 0, 0)
    end
    frame.BorderSizePixel = 0

    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(1, 0)
    corner.Parent = frame

    frame.Parent = gui
    indicatorFrame = frame
end

local function ensureIndicator()
    -- Check if indicator exists and is still in a valid GUI
    if indicatorFrame and indicatorFrame.Parent and indicatorFrame.Parent:IsDescendantOf(lplr) then
        return -- all good
    end
    -- Otherwise recreate
    createIndicator()
end

-- Listen for PlayerGui being added (if game recreates it after death)
local function watchPlayerGui()
    if playerGuiConn then playerGuiConn:Disconnect() end
    playerGuiConn = lplr:GetPropertyChangedSignal("Parent"):Connect(function()
        if lplr.Parent == nil then return end
        ensureIndicator()
    end)
    -- Also directly on CharacterAdded (common respawn trigger)
    lplr.CharacterAdded:Connect(function()
        ensureIndicator()
    end)
    -- Initial creation
    ensureIndicator()
end

----------------------------------------------------------------
-- CLEANUP on executor stop / teleport
----------------------------------------------------------------
local function fullCleanup()
    -- Disable hitboxes if active (removes all parts & connections)
    if hitboxesActive and CONFIG.HitboxMode == 'Player' then
        disableHitboxes()
    end
    -- Remove indicator GUI
    if indicatorFrame and indicatorFrame.Parent then
        indicatorFrame.Parent:Destroy()
    end
    if playerGuiConn then playerGuiConn:Disconnect() end
    -- Optional: attempt to restore reach constant to default?
    local cc = getCombatConstant()
    if cc and cc.RAYCAST_SWORD_CHARACTER_DISTANCE then
        pcall(function()
            cc.RAYCAST_SWORD_CHARACTER_DISTANCE = 14.4
        end)
    end
    print('[HitBoxes+Reach] Cleaned up')
end

-- Hook into RunService (most executors support Stopped)
local RunService = game:GetService('RunService')
RunService.Stopped:Connect(fullCleanup)
-- Also handle teleport (game binds to close)
game:GetService("TeleportService"):GetPropertyChangedSignal("TeleportRequest"):Connect(fullCleanup)

----------------------------------------------------------------
-- STARTUP (always on)
----------------------------------------------------------------
startHitboxes()
startReach()

-- Start the persistent dot watcher
watchPlayerGui()

-- Toggle key: Backquote (`) — the key above Tab
UserInputService.InputBegan:Connect(function(input, gameProcessed)
    if gameProcessed then return end
    if input.KeyCode == Enum.KeyCode.Backquote then
        toggleHitboxes()
    end
end)

if getgenv then
    getgenv().HitboxReach = {
        config = CONFIG,
        refresh = refresh,
        toggle = toggleHitboxes,
        enabled = function() return hitboxesActive end,
    }
end

print('[HitBoxes+Reach] Ready — Hitbox: ' .. CONFIG.HitboxMode .. ' (Backquote toggles if Player mode) | SwordReach: ' .. tostring(CONFIG.SwordReach))
print('                  Backquote (`) toggles hitboxes – green = ON, red = OFF, yellow = Sword mode (no toggle)')
